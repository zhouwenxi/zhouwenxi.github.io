AsyncTask异步解析

发表于 2018-04-11 | 分类于 Android | 阅读次数: |

简介

AsyncTask是对Handler和线程池的封装，为了避免ANR问题。

使用

i:最简化状态

 AsyncTask task=new AsyncTask() {
            @Override
            protected Object doInBackground(Object[] objects) {
                Log.e("qishui","调用doInBackground方法...");
                return null;
            }
        };
task.execute();

以上是最简单的使用方法，在doInBackground方法中运行耗时操作。

ii:一般状态

 AsyncTask task=new AsyncTask<Object,Integer,String>() {
            @Override
            protected void onPreExecute() {
                Log.e("qishui2","调用onPreExecute方法...");
            }
            @Override
            protected String doInBackground(Object[] objects) {
                Log.e("qishui2","调用doInBackground方法..."+objects[0]);
                return "Hello world !";
            }
            @Override
            protected void onPostExecute(String s) {
                Log.e("qishui2","调用onPostExecute方法... "+s);
            }
        };
task.execute("a dog");

关键点解析

task.execute("a dog");中传入的参数是作为doInBackground方法的第一个参数；
doInBackground方法的返回结果作为onPostExecute方法参数，Hello world !
其中onPreExecute方法是执行在UI线程中，一般用于开始耗时之前的提示状态信息；
doInBackground方法就是运行在子线程中，用来执行耗时动作；
onPostExecute方法也是在UI线程中，它的作用就是接受结果并做出相应处理。
而execute方法接受的也是可变参数类型，这个所代表的的就是doInBackground
的可变参数接受数据。如果execute方法传入多个数据，onPreExecute,onPostExecute 执行一次。

重要类型解析

第一个参数类型Object，所对应的是doInBackground方法可变参数类型；
第二个参数类型Integer，所对应的是onProgressUpdate方法可变参数类型；
第三个参数类型String，所对应的是doInBackground方法返回数据类型，也是onPostExecute方法参数类型

注意

新建和执行需要在UI线程,并且每个实例只能执行一次

怎么才能确认在Ui线程中调用呢？

实际上可以在application中获取线程id，其实就是代表UI线程id,在使用之前判断当前是否是UI线程就可以区分开来。

一个综合实例

/**
 * @author Zhou
 * Created on 2018/4/11 16:33.
 * Email:qishuichixi@163.com
 * Desc:异步任务描述
 */
public class FileAsyncTask {

    private static final String KEY_TAG = "qishui";
    private View[] views;

    public FileAsyncTask(View[] views, Object... message) {
        this.views = views;
        init(message);
    }

    /**
     * 开始异步任务
     */
    private void init(Object... message) {

        @SuppressLint("StaticFieldLeak")
        AsyncTask asyncTask = new AsyncTask<Object, Integer, String>() {

            @Override
            protected void onPreExecute() {

                TextView tv = (TextView) views[0];
                tv.setText("开始执行 onPreExecute 在UI线程");
                Log.e(KEY_TAG, "开始执行 onPreExecute 在UI线程");
            }

            @Override
            protected String doInBackground(Object... objects) {

                int length = objects.length;
                String[] value = new String[length];
                Log.e(KEY_TAG, "开始执行 doInBackground 在子线程");

                for (int i = 0; i < length; i++) {
                    value[i] = "" + objects[i];
                    Log.e(KEY_TAG, "参数传递数据:" + value[i]);
                }

                //更新UI
                Integer[] integers = new Integer[length];
                for (int i = 0; i < 97; i++) {
                    integers[0] = i;
                    if (length >= 2) {
                        integers[1] = i + 3;
                    }

                    //睡一秒更新
                    try {
                        Thread.sleep(1000);
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }

                    publishProgress(integers);
                }


                return "Hello World !";
            }

            @Override
            protected void onProgressUpdate(Integer... values) {

                Log.e(KEY_TAG, "开始执行 onProgressUpdate 在UI线程");

                TextView tv1 = (TextView) views[1];
                tv1.setText("更新中:" + values[0] + "%");

                TextView tv2 = (TextView) views[2];
                tv2.setText("更新中:" + values[1] + "%");


                ProgressBar pb1 = (ProgressBar) views[3];
                ProgressBar pb2 = (ProgressBar) views[4];

                pb1.setProgress(values[0]);
                Log.e(KEY_TAG, "更新第一条进度:" + values[0]);

                if (values.length >= 2) {
                    pb2.setProgress(values[1]);
                    Log.e(KEY_TAG, "更新第二条进度:" + values[1]);
                }

            }


            @Override
            protected void onPostExecute(String s) {

                TextView tv = (TextView) views[0];
                tv.setText("执行结束 onPostExecute 在UI线程:" + s);
                Log.e(KEY_TAG, "执行结束 onPostExecute 在UI线程:" + s);
            }
        };
        asyncTask.execute(message);

    }

}

主要实体

/**
 * @author Zhou
 *         Created on 2018/4/11 13:54.
 *         Email:qishuichixi@163.com
 *         Desc:
 */
public class FileReadAsyncTask {

    private CallBack callBack;

    public interface CallBack {
        void success(List list);
    }
    public void setCallBack(CallBack callBack) {
        this.callBack = callBack;
    }

    public FileReadAsyncTask(String msg) {
        init(msg);
    }

    @SuppressLint("StaticFieldLeak")
    private void init(final String msg) {

        AsyncTask task = new AsyncTask<Object, Integer, List>() {
            @Override
            protected List doInBackground(Object... objects) {
                return FileUtils.readMessageByLine(msg);
            }

            @Override
            protected void onPostExecute(List list) {

                if (list.size() > 0) {
                    if (callBack != null) {
                        callBack.success(list);
                    }
                }
            }
        };
        task.execute(msg);
    }

}

读取信息行为单位

/**
 * 读取信息行为单位
 *
 * @param filePath
 * @return
 */
public static List<String> readMessageByLine(String filePath) {

    List list = new ArrayList<String>();
    File file = new File(StringUtils.addString(FileUtils.getSDPath(), File.separator, filePath));
    if (!file.exists()) {
        return list;
    }

    try {
        String temp = "";
        BufferedReader reader = new BufferedReader(new FileReader(file));
        while ((temp = reader.readLine()) != null) {
            list.add(temp);
        }
        reader.close();
    } catch (Exception e) {
        LogUtils.file("读取信息行为单位发生异常!");
        e.printStackTrace();
    }

    return list;

}
